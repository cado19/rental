CREATE TABLE `kisuzi-rental`.`customer_details` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `first_name` VARCHAR(45) NULL,
  `last_name` VARCHAR(60) NULL,

  `email` VARCHAR(100) NULL,
  `date_of_birth` DATETIME NULL,
